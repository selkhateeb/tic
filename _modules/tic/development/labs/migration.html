    rpc = datastore.GetRpcFromKwargs({}) 
    keys, multiple = datastore.NormalizeAndTypeCheckKeys(keys) 
    entities = None 
    try: 
        entities = datastore.Get(keys, rpc=rpc) 
    except datastore_errors.EntityNotFoundError: 
        assert not multiple 
    return entities
